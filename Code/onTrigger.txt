void OnTriggerEnter (Collider collider) {
	//Bei Ber端hrung eines Buschen wird die Kugel um 50% verlangsamt
	if ( collider.CompareTag ( "Busch" ) ) {
		//maxspeed 端ber die multiplier Variable auf 50% setzen
		speedMultiplier = stSphere.SlowDownSpeedTo;
		Vector3 oppositeForce = -rigSphere.velocity;
		oppositeForce *= 0.5f;
		//Entgegen der Bewegungsrichtung Schub aus端ben
		rigSphere.AddForce ( oppositeForce, ForceMode.VelocityChange );
	}
}

void OnTriggerExit ( Collider collider ) {
	//Beim Austreten aus dem Buschcollider den -speedMultiplier- zur端cksetzen
	if ( collider.CompareTag ( "Busch" ) ) {
		speedMultiplier = 1f;
	}
}
